📌 Section 3: Async/Await Patterns

public async Task CreateResourceAsync(string name) { try { var resource = await _client.CreateResourceAsync(name); _logger.LogInformation("Created resource: {Name}", name); return resource; } catch (Exception ex) { _logger.LogError(ex, "Failed to create resource"); throw; } } // Usage var result = await CreateResourceAsync("myResource");
Async Best Practice: Always use async/await in I/O operations, avoid .Result() blocking.

📌 Section 4: HTTP Clients

public class ApiService { private readonly HttpClient _httpClient; public ApiService(HttpClient httpClient) { _httpClient = httpClient; } public async Task GetAsync(string url) { var response = await _httpClient.GetAsync(url); response.EnsureSuccessStatusCode(); return await response.Content.ReadAsAsync(); } }

📌 Section 5: Dependency Injection

var services = new ServiceCollection(); // Register services services.AddSingleton(config); services.AddScoped(); services.AddHttpClient(); // Build provider var provider = services.BuildServiceProvider(); var resourceMgr = provider.GetRequiredService();
DI Pattern: Always inject dependencies, avoid static instantiation.

📌 Section 6: Error Handling

try { await _client.DeployAsync(template); } catch (Azure.RequestFailedException ex) when (ex.Status == 409) { _logger.LogWarning("Resource already exists"); } catch (Azure.RequestFailedException ex) { _logger.LogError(ex, "Deployment failed with status {Status}", ex.Status); throw; }
